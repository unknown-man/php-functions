function insert($table,$x){
echo "INSERT INTO $table"."(";
$r = count($x)-1;
$p = array_keys($x);
for ($i=0; $i < $r; $i++) {
  echo $p[$i];
  if ($i <= $r-2) {
    echo ",";
  }
}
echo ") VALUES(";
for ($i=0; $i < $r; $i++) {
  echo base64_encode($x[$p[$i]]);
  if ($i <= $r-2) {
    echo ",";
  }
}
echo ")";
}
